**To calculate an aspect:**
1. Convert both planet positions to absolute degrees (0-360)

3. Check if within orb of any aspect angle

```python
# Natal chart positions (absolute degrees)

    'Chiron': 158,   # 8° Virgo

    'Midheaven': 338  # 8° Pisces

# Sign conversion helper
def to_absolute(sign, degree):
    signs = ['Aries','Taurus','Gemini','Cancer','Leo','Virgo',
             'Libra','Scorpio','Sagittarius','Capricorn','Aquarius','Pisces']
    return signs.index(sign) * 30 + degree

# Aspect check
def check_aspect(transit_deg, natal_deg):
    diff = abs(transit_deg - natal_deg)
    if diff > 180: diff = 360 - diff
    aspects = {
        'Conjunction': (0, 8),
        'Opposition': (180, 8),
        'Trine': (120, 8),
        'Square': (90, 7),
        'Sextile': (60, 6),
        'Inconjunct': (150, 3),
        'Sesquiquadrate': (135, 3),
        'Semi-square': (45, 3),
        'Semi-sextile': (30, 2),
        'Quintile': (72, 2)
    }
    for name, (angle, orb) in aspects.items():
        if abs(diff - angle) <= orb:
            return name, abs(diff - angle)
    return None, None
```
